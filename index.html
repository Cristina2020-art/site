Aici veți găsi lecții despre limbajul de programare C++:

Introducere în C++

C++ este un limbaj de programare de nivel înalt, dezvoltat de Bjarne Stroustrup în 1979 la Bell Labs.

Este cunoscut pentru suportul său pentru programarea orientată pe obiecte (OOP).

Variabile și tipuri de date

În C++, variabilele sunt utilizate pentru a stoca date.

Fiecare variabilă trebuie să aibă un tip de date specific, cum ar fi int, float, char, etc.

Structuri de control

C++ suportă diverse structuri de control, cum ar fi if, else, for, while, și switch, pentru a controla fluxul de execuție al programului.

Proiect: Calculator simplu

Un calculator simplu care poate efectua operații de adunare, scădere, înmulțire și împărțire.


#include 
using namespace std;

int main() {
    char op;
    float num1, num2;
    cout << "Introduceți operatorul (+, -, *, /): ";
    cin >> op;
    cout << "Introduceți două numere: ";
    cin >> num1 >> num2;

    switch(op) {
        case '+':
            cout << num1 << " + " << num2 << " = " << num1 + num2;
            break;
        case '-':
            cout << num1 << " - " << num2 << " = " << num1 - num2;
            break;
        case '*':
            cout << num1 << " * " << num2 << " = " << num1 * num2;
            break;
        case '/':
            if(num2 != 0)
                cout << num1 << " / " << num2 << " = " << num1 / num2;
            else
                cout << "Împărțirea la zero nu este permisă.";
            break;
        default:
            cout << "Operator invalid.";
            break;
    }

    return 0;
}

Proiect: Joc de ghicit numere

Un joc simplu în care utilizatorul trebuie să ghicească un număr generat aleatoriu de computer.


#include 
#include 
#include 
using namespace std;

int main() {
    srand(time(0));
    int numarSecret = rand() % 100 + 1;
    int incercare;
    cout << "Ghici numărul (între 1 și 100): ";
    do {
        cin >> incercare;
        if (incercare > numarSecret)
            cout << "Numărul este prea mare. Încearcă din nou: ";
        else if (incercare < numarSecret)
            cout << "Numărul este prea mic. Încearcă din nou: ";
    } while (incercare != numarSecret);

    cout << "Felicitări! Ai ghicit numărul." << endl;
    return 0;
}

Aplicație: Manager de sarcini

O aplicație simplă pentru gestionarea sarcinilor zilnice.


// Cod simplificat pentru un manager de sarcini
#include 
#include 
#include 
using namespace std;

void afisareMeniu() {
    cout << "1. Adaugă sarcină" << endl;
    cout << "2. Afișează sarcini" << endl;
    cout << "3. Ieșire" << endl;
}

int main() {
    vector sarcini;
    int optiune;
    string sarcina;
    do {
        afisareMeniu();
        cin >> optiune;
        cin.ignore();
        switch(optiune) {
            case 1:
                cout << "Introduceți sarcina: ";
                getline(cin, sarcina);
                sarcini.push_back(sarcina);
                break;
            case 2:
                cout << "Sarcinile tale:" << endl;
                for (int i = 0; i < sarcini.size(); i++) {
                    cout << i + 1 << ". " << sarcini[i] << endl;
                }
                break;
            case 3:
                cout << "La revedere!" << endl;
                break;
            default:
                cout << "Opțiune invalidă. Încercați din nou." << endl;
                break;
        }
    } while (optiune != 3);
    return 0;
}
